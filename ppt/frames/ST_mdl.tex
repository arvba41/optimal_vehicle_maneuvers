                F_{y(i)} &= F_{y0(i)}\left(\alpha_i\right)\sqrt{1 - \left(\frac{F_{x(i)}}{F_{x0(i)}^{\text   {max}}}\right)}

            \begin{align*}
                \dot F_{x(i)} = \frac{1}{\tau_f}\left(F_{x(i)} - F_{x(i)}^*\right)
            \end{align*}
        \end{column}
